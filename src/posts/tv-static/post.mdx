```javascript
const c = document.getElementById('canvas');
const ctx = c.getContext('2d');

const draw = () => {
	ctx.clearRect(0, 0, c.width, c.height);
	ctx.fillStyle = "#000";

	for (let i = 0; i < 3e4; i++) {
		ctx.fillRect(Math.random() * c.height, Math.random() * c.width, 3, 3);
	}

	setTimeout(() => requestAnimationFrame(draw), 30);
}

draw();
```
